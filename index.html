Exemplo de código não otimizado

function testFunc() {
   var text = "";
    $('div:visible').each(function(){
        text += $(this).text();
    });

    $('div:visible').css('color','red');
}

console.time('demo timer');

var i;

for( i=0; i<1000; i++ ) {
    testFunc();
}

console.timeEnd('demo timer');

Exemplo otimizado

function testFunc() {
   var text = "";
    var divs = $('div:visible');
    divs.each(function(){
        text += this.innerText;
    });
    // outros códigos
    divs.css('color','red');
}

console.time('demo timer');

var i;

for( i=0; i<1000; i++ ) {
    testFunc();
}

console.timeEnd('demo timer');
